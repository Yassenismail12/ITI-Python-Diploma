Sorting & Reversing

# In-place operations - O(n log n)
lst.sort()              # Ascending
lst.sort(reverse=True)    # Descending
lst.sort(key=lambda x: x%2)# Custom key
lst.reverse()           # O(n) - reverse order

# Return new list
sorted_lst = sorted(lst)  # O(n log n)
reversed_lst = lst[::-1]  # O(n) - slice trick

üéØ List Tricks

# List comprehension - Pythonic and fast
squares = [x**2 for x in range(10)]
evens = [x for x in lst if x % 2 == 0]

# Multiple assignment
a, b, c = [1, 2, 3]
*head, tail = [1, 2, 3, 4]  # head=[1,2,3], tail=4

# Flattening nested lists
nested = [[1, 2], [3, 4], [5]]
flat = [item for sublist in nested for item in sublist]

# Remove duplicates while preserving order
unique = list(dict.fromkeys(lst))

# Rotate list
def rotate_right(lst, k):
    k %= len(lst)
    return lst[-k:] + lst[:-k]

üéØ Dictionary Tricks

# Dictionary comprehension

filtered = {k: v for k, v in d.items() if v > 2}

# Merging dictionaries (Python 3.9+)

# Default dictionaries

# Counter for counting

# Reverse dictionary

# Group by key

grouped = {k: [item[1] for item in group]
           for k, group in groupby(data, key=lambda x: x[0])}

üöÄ Optimization Tricks

# Use collections.deque for frequent insertions at both ends
from collections import deque
d = deque([1, 2, 3])
d.appendleft(0)  # O(1) vs O(n) for list.insert(0, x)

# Use sets for membership testing
# Instead of: if item in long_list  # O(n)
# Use: if item in set(long_list)    # O(1) after conversion

# Use dict.get() instead of checking keys
# Instead of:
if key in dict:
    value = dict[key]
else:
    value = default
# Use: value = dict.get(key, default)

# Use list comprehensions instead of loops (faster)
# Instead of:
result = []
for x in range(10):
    if x % 2 == 0:
        result.append(x**2)
# Use: result = [x**2 for x in range(10) if x % 2 == 0]

# Use enumerate() instead of manual indexing
# Instead of:
for i in range(len(lst)):
    print(i, lst[i])
# Use:
for i, value in enumerate(lst):
    print(i, value)

üì¶ Advanced Collections

collections.deque

from collections import deque

# Double-ended queue - O(1) operations at both ends
dq = deque([1, 2, 3])
dq.appendleft(0)     # Add to left
dq.append(4)         # Add to right  
dq.popleft()         # Remove from left
dq.pop()             # Remove from right
dq.rotate(1)         # Rotate right

collections.Counter

from collections import Counter

# Count hashable objects
c = Counter('hello world')  # Counter({'l': 3, 'o': 2, ...})
c.most_common(3)            # [('l', 3), ('o', 2), ('h', 1)]
c.update('hello')           # Add more counts

collections.OrderedDict

from collections import OrderedDict

# Dictionary that remembers insertion order (built-in in Python 3.7+)
od = OrderedDict([('a', 1), ('b', 2)])
od.move_to_end('a')        # Move key to end
